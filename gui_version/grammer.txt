expr    : KEYWORD:VAR IDENTIFIER EQ expr
            : comp-expr ((KEYWORD:AND|KEYWORD:OR) comp-expr)*

comp-expr   : NOT comp-expr
            : arith-expr ((EE|LT|GT|LTE|GTE) arith-expr)*

arith-expr  : term ((جمع|طرح) term)*

term    : factor ((ضرب|قسمة) factor)*

factor  : (PLUS|MINUS) factor
            : power

power   : call (POW factor)*

call    : atom(LPAREN(expr(COMMA expr)*)? RPAREN)

atom 				: INT|FLOAT|STRING|IDENTIFIER
						: LPAREN expr RPAREN
						: list-expr
						: if-expr
						: for-expr
						: while-expr
						: func-def

list-expr		: LSQUARE (expr (COMMA expr)*)? RSQUARE

if-expr			: KEYWORD:IF expr KEYWORD:THEN expr
							(KEYWORD:ELIF expr KEYWORD:THEN expr)*
							(KEYWORD:ELSE expr)?

for-expr		: KEYWORD:لكل IDENTIFIER EQ expr KEYWORD:إلى expr 
							(KEYWORD:خطوة expr)? KEYWORD:ثم expr

while-expr	: KEYWORD:بينما expr KEYWORD:ثم expr

func-def		: KEYWORD:FUN IDENTIFIER?
							LPAREN (IDENTIFIER (COMMA IDENTIFIER)*)? RPAREN
							ARROW expr

COMMENT         : '#' <any character>* NEWLINE          # Inline comment
                | '##' <any character>* '##'             # Multiline comment
